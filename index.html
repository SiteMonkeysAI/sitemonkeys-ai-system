let sessionVault = null; // Store vault in session let sessionActive = false; let sessionCostTotal = 0; // Track session costs let dailyCostTotal = parseFloat(localStorage.getItem('dailyCost') || '0'); let dailyDate = localStorage.getItem('dailyDate') || new Date().toDateString(); // Reset daily cost if new day if (dailyDate !== new Date().toDateString()) { dailyCostTotal = 0; dailyDate = new Date().toDateString(); localStorage.setItem('dailyCost', '0'); localStorage.setItem('dailyDate', dailyDate); } function updateCostDisplay(costInfo) { if (costInfo && costInfo.estimated_cost) { const messageCost = parseFloat(costInfo.estimated_cost); sessionCostTotal += messageCost; dailyCostTotal += messageCost; // Save daily cost localStorage.setItem('dailyCost', dailyCostTotal.toString()); // Update display let costDisplay = document.getElementById('costDisplay'); if (!costDisplay) { costDisplay = document.createElement('div'); costDisplay.id = 'costDisplay'; costDisplay.style.cssText = ` position: fixed; top: 10px; right: 10px; background: rgba(0,0,0,0.8); color: white; padding: 10px; border-radius: 8px; font-size: 12px; z-index: 1000; `; document.body.appendChild(costDisplay); } costDisplay.innerHTML = ` üí∞ Cost Tracking
This message: ${messageCost.toFixed(4)}
Session total: ${sessionCostTotal.toFixed(4)}
Today total: ${dailyCostTotal.toFixed(4)}
Tokens: ${costInfo.total_tokens || 'N/A'} `; } } async function initializeSession() { if (sessionVault) { console.log('Session vault already loaded'); return sessionVault; } showStatus('Loading SiteMonkeys business intelligence vault...', 'loading'); try { const response = await fetch('/api/load-vault'); const data = await response.json(); if (data.success) { sessionVault = data.memory; // Store in session sessionActive = true; const tokenCount = data.token_estimate || 'Unknown'; showStatus(`‚úÖ Session vault loaded (${tokenCount} tokens) - Ready for validation`, 'success'); console.log('Vault loaded and cached for session'); return sessionVault; } else { throw new Error(data.error || 'Failed to load vault'); } } catch (error) { console.error('Vault loading error:', error); showStatus('‚ùå Failed to load business intelligence vault', 'error'); throw error; } } async function sendMessage() { const input = document.getElementById('userInput'); const message = input.value.trim(); if (!message) return; // Initialize session vault if not loaded if (!sessionVault) { try { await initializeSession(); } catch (error) { return; // Exit if vault loading fails } } // Add user message to chat addMessage(message, 'user'); input.value = ''; // Show AI thinking const thinkingDiv = addMessage('üß† Analyzing against SiteMonkeys business intelligence...', 'assistant', true); try { const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ message: message, vault_memory: sessionVault // Use cached vault }) }); const data = await response.json(); // Remove thinking message thinkingDiv.remove(); if (data.success) { addMessage(data.response, 'assistant'); // Update cost tracking if (data.cost_info) { updateCostDisplay(data.cost_info); } } else { addMessage(`‚ùå Error: ${data.error}`, 'assistant'); } } catch (error) { thinkingDiv.remove(); addMessage('‚ùå Connection error.
